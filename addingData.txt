/*
*	здесь будет скрипт, который с помощью curl
*	будет инициализировать базу данных
*	таким образом мы можем избежать 
*	ошибок с кодировками
*	194.58.100.50 <-=-> 194.58.100.50
*/

curl 194.58.100.50/newTown -d "name=Ufa&russianName=Уфа"
curl 194.58.100.50/newTown -d "name=Uchaly&russianName=Учалы"
curl 194.58.100.50/newTown -d "name=Sibay&russianName=Сибай"

curl 194.58.100.50/newDirection -d "source=1&destination=2"
curl 194.58.100.50/newDirection -d "source=1&destination=3"
curl 194.58.100.50/newDirection -d "source=2&destination=1"
curl 194.58.100.50/newDirection -d "source=3&destination=1"

curl 194.58.100.50/newTime -d "name=00:00-03:00"
curl 194.58.100.50/newTime -d "name=03:00-06:00"
curl 194.58.100.50/newTime -d "name=06:00-09:00"
curl 194.58.100.50/newTime -d "name=09:00-12:00"
curl 194.58.100.50/newTime -d "name=12:00-15:00"
curl 194.58.100.50/newTime -d "name=15:00-18:00"
curl 194.58.100.50/newTime -d "name=18:00-21:00"
curl 194.58.100.50/newTime -d "name=21:00-00:00"
